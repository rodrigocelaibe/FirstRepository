.delete - eliminar (produce efecto)
.push - agregar (produce efecto)
.size - consultar tamaño (no produce efecto)
.include? - está? (no produce efecto)

.all? - todos? (no produce efecto)
.any? - alguno? (no produce efecto)
.count - contar elementos (no produce efecto)
.sum - sumatoria elementos (no produce efecto)
.map - "modificar" colección (no produce efecto)
.each - "modificar" colección (produce efecto)
